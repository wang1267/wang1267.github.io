/* 循迹传感器读取与位置偏差计算 */
uint8_t readSensors(void) {
    uint8_t pattern = 0;
    for (int i = 0; i < 4; i++) {
        pattern |= (GPIO_ReadInputDataBit(sensor_port[i], sensor_pin[i]) << i);
    }
    return pattern;
}

/* PID 控制器 — 根据偏差计算差速 */
float PID_Compute(float error) {
    integral += error * DT;
    derivative = (error - prev_error) / DT;
    prev_error = error;
    return Kp * error + Ki * integral + Kd * derivative;
}

/* 电机 PWM 输出 — 左轮/右轮独立调速 */
void motorControl(float correction) {
    int16_t left  = BASE_SPEED - correction;
    int16_t right = BASE_SPEED + correction;
    TIM_SetCompare1(TIM2, clamp(left,  0, MAX_PWM));
    TIM_SetCompare2(TIM2, clamp(right, 0, MAX_PWM));
}
